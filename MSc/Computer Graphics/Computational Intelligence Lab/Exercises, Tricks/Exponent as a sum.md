$$
\begin{align}
	\prod^{k}_{j=1}\pi_j^{\alpha_j-1} \prod^{n}_{i=1} \pi_j^{x_{ij}} &= \prod^{k}_{j=1} \left( \pi_j^{\alpha_j - 1} \pi_j^{\sum^{n}_{i=1}x_{ij}} \right) \\
	&= \prod^{k}_{j=1} \pi_j^{\alpha_j - 1 + m_j}
\end{align}
$$
Where $m_j = \sum^{n}_{i=1}x_{ij}$.
